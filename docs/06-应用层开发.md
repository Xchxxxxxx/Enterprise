## 6.1 泛型 CRUD（零代码增删改查）

继承 `CrudAppService`，**一行代码都不用写**，自动获得完整 CRUD：

```csharp
using AutoMapper;
using EfCore.Enterprise.Application.Crud;
using EfCore.Enterprise.Domain.Interfaces;
using MyApp.Contracts.News.Dtos;
using MyApp.Domain.Entities;

namespace MyApp.Application.News.Services;

public class NewsService : CrudAppService<News, NewsDto, CreateNewsDto, UpdateNewsDto>
{
    public NewsService(ISuperRepository<News> repository, IMapper mapper)
        : base(repository, mapper)
    {
    }
}
```

**自动获得的方法**：

| 方法 | 签名 | 说明 |
|------|------|------|
| `GetPageAsync` | `Task<PagedResult<TDto>> GetPageAsync(PagedRequest)` | 分页查询 |
| `GetByIdAsync` | `Task<TDto?> GetByIdAsync(long id)` | 根据ID查询 |
| `CreateAsync` | `Task<TDto> CreateAsync(TCreateDto dto)` | 创建 |
| `UpdateAsync` | `Task<TDto> UpdateAsync(long id, TUpdateDto dto)` | 更新 |
| `DeleteAsync` | `Task<bool> DeleteAsync(long id)` | 软删除 |

### 重写过滤条件

```csharp
public class MyNewsService : CrudAppService<News, NewsDto, CreateNewsDto, UpdateNewsDto>
{
    public MyNewsService(ISuperRepository<News> repository, IMapper mapper)
        : base(repository, mapper) { }

    // 只返回已发布的新闻
    protected override IQueryable<News> ApplyFilter(IQueryable<News> query, PagedRequest request)
    {
        return query.Where(x => x.Status == "Published");
    }
}
```

### 重写单个方法

```csharp
public class MyNewsService : CrudAppService<News, NewsDto, CreateNewsDto, UpdateNewsDto>
{
    public override async Task<TDto> CreateAsync(TCreateDto dto)
    {
        // 创建前自定义逻辑
        await ValidateTitleAsync(dto.Title);
        return await base.CreateAsync(dto);
    }

    public override async Task<TDto> UpdateAsync(long id, TUpdateDto dto)
    {
        // 更新前检查权限
        if (!await _permissionService.CanEditNewsAsync(id))
            throw new ForbiddenException("无权编辑此新闻");

        return await base.UpdateAsync(id, dto);
    }
}
```

## 6.2 自定义业务服务

```csharp
using System.Linq.Expressions;
using AutoMapper;
using EfCore.Enterprise.Application.Services;
using EfCore.Enterprise.Domain.Interfaces;
using EfCore.Enterprise.Shared.Exceptions;
using EfCore.Enterprise.Shared.Models;
using MyApp.Application.Products.Queries;
using MyApp.Contracts.Products.Dtos;
using MyApp.Domain.Entities;

namespace MyApp.Application.Products.Services;

public class ProductService : BaseService
{
    private readonly ISuperRepository<Product> _repository;

    public ProductService(
        ISuperRepository<Product> repository,
        IUnitOfWork unitOfWork,
        IMapper mapper)
        : base(unitOfWork, mapper)
    {
        _repository = repository;
    }

    // ===== 创建 =====
    public async Task<ProductDto> CreateAsync(CreateProductDto dto, CancellationToken ct = default)
    {
        var product = new Product(dto.Name, dto.Price, dto.Stock);
        if (!string.IsNullOrWhiteSpace(dto.Description))
            product.ChangeDescription(dto.Description);

        await _repository.AddAsync(product, ct);
        await UnitOfWork.SaveChangesAsync(ct);

        return Mapper.Map<ProductDto>(product);
    }

    // ===== 更新价格 =====
    public async Task<ProductDto> UpdatePriceAsync(UpdateProductPriceDto dto, CancellationToken ct = default)
    {
        var product = await _repository.GetByIdAsync(dto.Id, ct);
        if (product is null)
            throw new NotFoundException($"产品 {dto.Id} 不存在");

        product.ChangePrice(dto.NewPrice);
        await UnitOfWork.SaveChangesAsync(ct);

        return Mapper.Map<ProductDto>(product);
    }

    // ===== 分页查询 =====
    public async Task<PagedResult<ProductDto>> GetPagedListAsync(
        GetProductListQuery query, CancellationToken ct = default)
    {
        Expression<Func<Product, bool>> predicate = x =>
            (string.IsNullOrWhiteSpace(query.Keyword) || x.Name.Contains(query.Keyword)) &&
            (!query.MinPrice.HasValue || x.Price >= query.MinPrice.Value) &&
            (!query.MaxPrice.HasValue || x.Price <= query.MaxPrice.Value);

        var request = new PagedRequest
        {
            PageIndex = query.PageIndex,
            PageSize = query.PageSize
        };

        var result = await _repository.GetPagedAsync(predicate, request, ct);

        return new PagedResult<ProductDto>(
            Mapper.Map<List<ProductDto>>(result.Items),
            result.TotalCount,
            result.PageIndex,
            result.PageSize);
    }

    // ===== 查询单个 =====
    public async Task<ProductDto?> GetByIdAsync(long id, CancellationToken ct = default)
    {
        var product = await _repository.GetByIdAsync(id, ct);
        return product is null ? null : Mapper.Map<ProductDto>(product);
    }

    // ===== 事务示例 =====
    public async Task TransferStockAsync(long fromId, long toId, int quantity)
    {
        var from = await _repository.GetByIdAsync(fromId)
            ?? throw new NotFoundException("源产品不存在");
        var to = await _repository.GetByIdAsync(toId)
            ?? throw new NotFoundException("目标产品不存在");

        from.ChangeStock(from.Stock - quantity);
        to.ChangeStock(to.Stock + quantity);

        await UnitOfWork.BeginTransactionAsync();
        try
        {
            await _repository.UpdateAsync(from);
            await _repository.UpdateAsync(to);
            await UnitOfWork.SaveChangesAsync();
            await UnitOfWork.CommitTransactionAsync();
        }
        catch
        {
            await UnitOfWork.RollbackTransactionAsync();
            throw;
        }
    }
}
```

## 6.3 AutoMapper 配置（BaseProfile）

```csharp
// 04-Application/Products/Mapping/ProductProfile.cs
using AutoMapper;
using EfCore.Enterprise.Application.Mapping;
using MyApp.Contracts.Products.Dtos;
using MyApp.Domain.Entities;

namespace MyApp.Application.Products.Mapping;

public class ProductProfile : BaseProfile
{
    protected override void Configure()
    {
        CreateMap<Product, ProductDto>();
        CreateMap<CreateProductDto, Product>()
            .ConstructUsing(dto => new Product(dto.Name, dto.Price, dto.Stock));
        CreateMap<UpdateProductDto, Product>()
            .ForAllMembers(opts => opts.Condition(
                (src, dest, srcMember) => srcMember != null));
    }
}
```

**BaseProfile 的优势**：

```csharp
// 传统方式（需要覆写构造函数）
public class ProductProfile : Profile
{
    public ProductProfile()
    {
        CreateMap<Product, ProductDto>();
    }
}

// 框架方式（只需实现 Configure() 方法）
public class ProductProfile : BaseProfile
{
    protected override void Configure()
    {
        CreateMap<Product, ProductDto>();
    }
}
```

## 6.4 FluentValidation 验证

```csharp
// 04-Application/Products/Validation/CreateProductDtoValidator.cs
using FluentValidation;
using MyApp.Contracts.Products.Dtos;

namespace MyApp.Application.Products.Validation;

public class CreateProductDtoValidator : AbstractValidator<CreateProductDto>
{
    public CreateProductDtoValidator()
    {
        RuleFor(x => x.Name)
            .NotEmpty().WithMessage("产品名称不能为空")
            .MaximumLength(200).WithMessage("产品名称不能超过200个字符");

        RuleFor(x => x.Price)
            .GreaterThan(0).WithMessage("价格必须大于0");

        RuleFor(x => x.Stock)
            .GreaterThanOrEqualTo(0).WithMessage("库存不能为负数");
    }
}
```

## 6.6 抛出异常

```csharp
// 业务异常
throw new AppException(ErrorCode.BusinessError, "订单金额不能为负数");
throw new BusinessException("订单金额不能为负数");

// 资源不存在
throw new NotFoundException("订单不存在");

// 未授权
throw new UnauthorizedException("请先登录");

// 无权限
throw new ForbiddenException("无权查看此订单");

// 参数验证失败
throw new ValidationException("订单号不能为空");
throw new ValidationException(new List<string> { "订单号不能为空", "金额必须大于0" });

// 数据锁定
throw new DataLockedException("订单已锁定，无法修改");

// 数据归档
throw new DataArchivedException("订单已归档，无法修改");

// 并发冲突
throw new ConcurrentConflictException("数据已被其他用户修改，请刷新后重试");

// 重复请求
throw new DuplicateRequestException("请勿重复提交");
```
