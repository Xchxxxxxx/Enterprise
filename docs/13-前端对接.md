## 13.1 API 基地址

```
开发环境: http://localhost:5000
```

```http
Authorization: Bearer {access_token}
Content-Type: application/json
Idempotency-Key: {uuid}  (可选，用于幂等)
```

## 13.4 TypeScript / Axios 封装

```typescript
// api/types.ts
interface ApiResponse<T = unknown> {
  success: boolean;
  data?: T;
  code?: string;
  message?: string;
  traceId?: string;
}

interface PagedData<T> {
  items: T[];
  totalCount: number;
  pageIndex: number;
  pageSize: number;
  totalPages: number;
  hasPreviousPage: boolean;
  hasNextPage: boolean;
}

// api/client.ts
import axios, { AxiosInstance, AxiosRequestConfig } from 'axios';

class ApiClient {
  private client: AxiosInstance;

  constructor(baseURL: string) {
    this.client = axios.create({
      baseURL,
      timeout: 30000,
      headers: { 'Content-Type': 'application/json' }
    });

    // 请求拦截器：自动附加 Token
    this.client.interceptors.request.use(config => {
      const token = localStorage.getItem('access_token');
      if (token) {
        config.headers.Authorization = `Bearer ${token}`;
      }
      return config;
    });

    // 响应拦截器：统一错误处理
    this.client.interceptors.response.use(
      response => response,
      error => {
        if (error.response?.status === 401) {
          // Token 过期，跳转登录
          window.location.href = '/login';
        }
        return Promise.reject(error);
      }
    );
  }

  async get<T>(url: string, params?: any): Promise<ApiResponse<T>> {
    const { data } = await this.client.get<ApiResponse<T>>(url, { params });
    return data;
  }

  async getPaged<T>(url: string, params?: any): Promise<ApiResponse<PagedData<T>>> {
    return this.get<PagedData<T>>(url, params);
  }

  async post<T>(url: string, body?: any): Promise<ApiResponse<T>> {
    const { data } = await this.client.post<ApiResponse<T>>(url, body);
    return data;
  }

  async put<T>(url: string, body?: any): Promise<ApiResponse<T>> {
    const { data } = await this.client.put<ApiResponse<T>>(url, body);
    return data;
  }

  async delete<T>(url: string): Promise<ApiResponse<T>> {
    const { data } = await this.client.delete<ApiResponse<T>>(url);
    return data;
  }
}

export const api = new ApiClient('http://localhost:5000');
```

## 13.5 实际调用示例

```typescript
// api/products.ts
import { api } from './client';

interface ProductDto {
  id: number;
  name: string;
  price: number;
  stock: number;
  description?: string;
}

interface CreateProductDto {
  name: string;
  price: number;
  stock: number;
  description?: string;
}

interface ProductListQuery {
  keyword?: string;
  minPrice?: number;
  maxPrice?: number;
  pageIndex?: number;
  pageSize?: number;
}

export const productApi = {
  // 分页查询
  getList: (query: ProductListQuery) =>
    api.getPaged<ProductDto>('/api/v1/products', query),

  // 查询单个
  getById: (id: number) =>
    api.get<ProductDto>(`/api/v1/products/${id}`),

  // 创建
  create: (dto: CreateProductDto) =>
    api.post<ProductDto>('/api/v1/products', dto),

  // 更新价格
  updatePrice: (id: number, dto: { id: number; newPrice: number }) =>
    api.put<ProductDto>(`/api/v1/products/${id}/price`, dto),

  // 删除
  delete: (id: number) =>
    api.delete(`/api/v1/products/${id}`)
};
```

## 13.6 SignalR 实时连接

```typescript
// 安装 npm install @microsoft/signalr
import { HubConnectionBuilder, LogLevel } from '@microsoft/signalr';

const connection = new HubConnectionBuilder()
  .withUrl('http://localhost:5000/hubs/notification', {
    accessTokenFactory: () => localStorage.getItem('access_token') || ''
  })
  .withAutomaticReconnect()
  .configureLogging(LogLevel.Information)
  .build();

// 接收消息
connection.on('ReceiveMessage', (message: string) => {
  console.log('收到消息:', message);
});

connection.on('NewOrder', (order: any) => {
  console.log('新订单:', order);
});

// 启动连接
await connection.start();
```

## 13.7 错误处理

```typescript
// 统一错误处理
function handleApiError(response: ApiResponse) {
  if (!response.success) {
    const code = response.code;
    const message = response.message;

    switch (code) {
      case 'NOT_FOUND':
        ElMessage.warning(message || '数据不存在');
        break;
      case 'FORBIDDEN':
        ElMessage.error('没有操作权限');
        break;
      case 'RATE_LIMIT':
        ElMessage.warning('操作太频繁，请稍后再试');
        break;
      case 'CONCURRENT_CONFLICT':
        ElMessage.warning('数据已被修改，请刷新后重试');
        break;
      case 'DUPLICATE_REQUEST':
        ElMessage.info('请勿重复提交');
        break;
      default:
        ElMessage.error(message || '操作失败');
    }
  }
}
```
