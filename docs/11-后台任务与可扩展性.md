## 11.1 后台任务基础

注入 `IBackgroundJobService`：

```csharp
[Injectable]
public class NotificationService : BaseService
{

    public NotificationService(IBackgroundJobService jobService)
```

## 11.2 立即执行

```csharp
// 参数为空的方法
await _jobService.EnqueueAsync(() => Console.WriteLine("任务开始执行"));

// 带参数的方法
await _jobService.EnqueueAsync(() => SendEmail(userId, "欢迎注册"));

// 带 CancellationToken 的方法
await _jobService.EnqueueAsync(() => ProcessOrderAsync(orderId, CancellationToken.None));
```

## 11.3 延迟执行

```csharp
// 延迟 5 分钟执行
await _jobService.ScheduleAsync(
    () => SendEmail(userId, "注册5分钟后提醒"),
    TimeSpan.FromMinutes(5));

// 延迟 1 小时执行
await _jobService.ScheduleAsync(
    () => GenerateReportAsync(reportId),
    TimeSpan.FromHours(1));
```

## 11.4 定时任务（Cron）

```csharp
// 在应用启动时注册
public void ConfigureJobs()
{
    // 每天凌晨 2 点清理日志
    _jobService.AddRecurring("clear-logs", () => ClearOldLogs(), "0 2 * * *");

    // 每 30 分钟同步数据
    _jobService.AddRecurring("sync-data", () => SyncExternalData(), "*/30 * * * *");

    // 每小时健康检查
    _jobService.AddRecurring("health-check", () => HealthCheck(), "0 */1 * * *");

    // 每周一凌晨 3 点生成周报
    _jobService.AddRecurring("weekly-report", () => GenerateWeeklyReport(), "0 3 * * 1");
}
```

## 11.5 Cron 表达式速查

| 表达式 | 含义 |
|--------|------|
| `* * * * *` | 每分钟 |
| `*/5 * * * *` | 每 5 分钟 |
| `0 * * * *` | 每小时整点 |
| `0 2 * * *` | 每天凌晨 2 点 |
| `0 9 * * 1-5` | 工作日早 9 点 |
| `0 0 1 * *` | 每月 1 号凌晨 |
| `0 0 * * 0` | 每周日凌晨 |
| `0 3 * * 1` | 每周一凌晨 3 点 |

## 11.6 事件驱动的后台任务

```csharp
// 领域事件发生 → 自动触发后台任务
public class OrderPaidEventHandler : INotificationHandler<OrderPaidDomainEvent>
{
    private readonly IBackgroundJobService _jobService;

    public OrderPaidEventHandler(IBackgroundJobService jobService)
    {
        _jobService = jobService;
    }

    public async Task Handle(OrderPaidDomainEvent e, CancellationToken ct)
    {
        // 支付成功后异步：
        // 1. 发送通知
        await _jobService.EnqueueAsync(() => SendPaymentNotification(e.OrderId));

        // 2. 扣减库存
        await _jobService.EnqueueAsync(() => DeductInventory(e.OrderId));

        // 3. 延迟 30 分钟发送评价提醒
        await _jobService.ScheduleAsync(
            () => SendReviewReminder(e.OrderId),
            TimeSpan.FromMinutes(30));
    }
}
```

## 11.8 插件系统

```csharp
// 框架支持热插拔插件
[Injectable]
public class PluginService : BaseService
{
    private readonly IPluginManager _pluginManager;

    public PluginService(IPluginManager pluginManager)
    {
        _pluginManager = pluginManager;
    }

    public async Task LoadPluginsAsync()
    {
        var plugins = await _pluginManager.LoadPluginsAsync("./plugins");
        foreach (var plugin in plugins)
        {
            await _pluginManager.InitializeAsync(plugin);
        }
    }
}
```

```csharp
    public PoolService(IObjectPoolService pool)
    {
        _pool = pool;
    }

    public async Task ProcessLargeDataAsync()
    {
        var buffer = await _pool.RentAsync<byte[]>(1024 * 1024); // 租用 1MB 缓冲区
        try
        {
            // 使用 buffer 处理数据
        }
        finally
        {
            await _pool.ReturnAsync(buffer); // 归还
        }
    }
}
```
